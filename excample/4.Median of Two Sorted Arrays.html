1.There are two sorted arrays nums1 and nums2 of size m and n respectively. Find the median of the two sorted arrays. The overall run time complexity should be O(log (m+n)).

1.有个两个排序好的数组nums1和nums2,长度分别为m和n,找到这两个数组的中位数,所有的排序次数必须为O(log(m+n))

Examples1:

			nums1 = [1, 3]
			nums2 = [2]

			The median is 2.0

Examples2:

			nums1 = [1, 2]
			nums2 = [3, 4]

			The median is (2 + 3)/2 = 2.5

Solution

			/**
			 * 找到两个数组的中位数
			 * @param {number[]} nums1
			 * @param {number[]} nums2
			 * @return {number}
			 */
			var findMedianSortedArrays = function(nums1, nums2) {
				var temArray = nums1.concat(nums2),
					temNum,
					medianNum,
					temLang = temArray.length;
				for(var i = 0;i < temLang;i++){
					for(x = 0;x < temLang-i-1;x++){
						if (temArray[x] > temArray[x+1]){
							temNum = temArray[x];
							temArray[x] = temArray[x+1];
							temArray[x+1] = temNum;
						}
					}
				};
				return temLang%2 == 0? (temArray[temLang/2 - 1]+temArray[temLang/2])/2 : (temArray[(temLang - 1)/2]);
			};
			findMedianSortedArrays([1,2],[5,8]);
