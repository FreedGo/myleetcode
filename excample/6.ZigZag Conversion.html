			/**
			 * 给定一个字符串,返回最长的回文字符串
			 * @param {string} str
			 * @return {string} str
			 */
			var longestPalindrome = function(s) {
				var newStr,revNewStr;
				for(var i =0 ; i < s.length ; i++){
					for(var x = 0;x <= i; x++ ){
						newStr = s.substring(x,s.length-i+x);
						revNewStr = newStr.split("").reverse().join("");
						if (newStr === revNewStr){
							return newStr;
						}
					}
				}
			};
